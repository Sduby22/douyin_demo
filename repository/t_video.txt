DROP TABLE IF EXISTS `t_video`;
CREATE TABLE t_video (
	vid INT AUTO_INCREMENT COMMENT '视频id',
	author VARCHAR(40) NOT NULL COMMENT '发布视频作者',
	player_url varchar(100) COMMENT '播放链接',
	cover_url varchar(100) COMMENT '播放路径',
	favorite_count int COMMENT '喜爱数',
              comment_count int COMMENT '评论数',
	is_favorite tinyint DEFAULT '0' COMMENT '是否喜爱：0-未喜爱，1-喜爱',
	PRIMARY KEY (vid)
) ENGINE=InnoDB DEFAULT CHARSET=utf8;
